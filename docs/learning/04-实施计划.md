from segment_anything import sam_model_registry, SamPredictor
import torch
import numpy as np
from .base import BaseModelManager

class SAM1Manager(BaseModelManager):
    """SAM1 æ¨¡å‹ç®¡ç†å™¨"""
    
    def load_model(self, checkpoint_path: str, model_type: str = "vit_b"):
        """åŠ è½½ SAM1 æ¨¡å‹"""
        print(f"æ­£åœ¨åŠ è½½ SAM1 {model_type} from {checkpoint_path}...")
        
        self.model = sam_model_registry[model_type](checkpoint=checkpoint_path)
        self.model.to(self.device)
        self.model.eval()
        
        self.predictor = SamPredictor(self.model)
        self.is_loaded = True
        
        print(f"âœ… SAM1 {model_type} åŠ è½½å®Œæˆ (device: {self.device})")
    
    def generate_embedding(self, image: np.ndarray) -> np.ndarray:
        """ç”Ÿæˆå›¾ç‰‡ Embedding"""
        if not self.is_loaded:
            raise RuntimeError("æ¨¡å‹æœªåŠ è½½")
        
        with torch.no_grad():
            self.predictor.set_image(image)
            embedding = self.predictor.get_image_embedding()
            return embedding.cpu().numpy()
    
    def segment(
        self,
        image: np.ndarray,
        points: np.ndarray,
        labels: np.ndarray
    ) -> Tuple[np.ndarray, np.ndarray]:
        """ç‚¹å‡»åˆ†å‰²"""
        if not self.is_loaded:
            raise RuntimeError("æ¨¡å‹æœªåŠ è½½")
        
        self.predictor.set_image(image)
        
        masks, scores, _ = self.predictor.predict(
            point_coords=points,
            point_labels=labels,
            multimask_output=True
        )
        
        return masks, scores

import numpy as np

def encode_mask_rle(mask: np.ndarray) -> str:
    """RLE ç¼–ç """
    pixels = mask.flatten()
    pixels = np.concatenate([[0], pixels, [0]])
    runs = np.where(pixels[1:] != pixels[:-1])[0] + 1
    runs[1::2] -= runs[::2]
    return ' '.join(str(x) for x in runs)

def decode_mask_rle(rle: str, shape: tuple) -> np.ndarray:
    """RLE è§£ç """
    runs = np.array([int(x) for x in rle.split()])
    starts = runs[0::2]
    lengths = runs[1::2]
    
    mask = np.zeros(shape[0] * shape[1], dtype=np.uint8)
    for start, length in zip(starts, lengths):
        mask[start:start+length] = 1
    
    return mask.reshape(shape)

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from typing import List
import numpy as np
import time

from ..models.registry import model_registry
from ..core.image_loader import load_image_from_url
from ..utils.rle import encode_mask_rle

router = APIRouter(prefix="/api", tags=["segment"])

class Point(BaseModel):
    x: float
    y: float
    type: int  # 0=èƒŒæ™¯, 1=å‰æ™¯

class SegmentRequest(BaseModel):
    image_url: str
    points: List[Point]
    model: str = "sam1_vit_b"

class SegmentResponse(BaseModel):
    mask_rle: str
    score: float
    time_ms: float
    model: str

@router.post("/segment", response_model=SegmentResponse)
async def segment(request: SegmentRequest):
    """çº¯åç«¯åˆ†å‰²"""
    start_time = time.time()
    
    try:
        # 1. åŠ è½½å›¾ç‰‡
        image = await load_image_from_url(request.image_url)
        
        # 2. è·å–/åŠ è½½æ¨¡å‹
        manager = model_registry.get(request.model)
        if not manager:
            manager = model_registry.load(request.model)
        
        # 3. å‡†å¤‡ç‚¹å‡»
        points = np.array([[p.x, p.y] for p in request.points])
        labels = np.array([p.type for p in request.points])
        
        # 4. åˆ†å‰²
        masks, scores = manager.segment(image, points, labels)
        
        # 5. é€‰æ‹©æœ€ä½³ mask
        best_idx = np.argmax(scores)
        best_mask = masks[best_idx]
        
        # 6. RLE ç¼–ç 
        mask_rle = encode_mask_rle(best_mask)
        
        elapsed = (time.time() - start_time) * 1000
        
        return SegmentResponse(
            mask_rle=mask_rle,
            score=float(scores[best_idx]),
            time_ms=elapsed,
            model=request.model
        )
    
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
